Welcome to Transaction Monitoring

Transaction monitoring enables developers to push tests created in a development environment leveraging CasperJS to a production environment and allow the application to be monitored in a consistent way. No more just checking if the page loads, does the application respond how a customer expects it to? 

casper.test.begin('Google test, find first h3 and select it', 1, function suite(test){
    casper.start('http://google.fr/');

    casper.thenEvaluate(function(term) {
        document.querySelector('input[name="q"]').setAttribute('value', term);
        document.querySelector('form[name="f"]').submit();
    }, 'CasperJS');

    casper.then(function() {
        // Click on 1st result link
        this.click('h3.r a');
    });

    casper.then(function() {
        test.assertUrlMatch("http://casperjs.org/")
    });
    
    casper.run(function(){
        test.done();
    });
});
